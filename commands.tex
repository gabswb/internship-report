% Set color of \refeq
\newcommand*{\SavedEqref}{}
\let\SavedEqref\eqref
\renewcommand*{\eqref}[1]{%
  \begingroup
    \hypersetup{
      linkcolor=blue,
      linkbordercolor=blue,
    }%
    \SavedEqref{#1}%
  \endgroup
}

    }%
